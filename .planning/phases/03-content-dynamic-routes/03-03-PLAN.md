Loader contract (Task 1 defines; routes consume):
```ts
// cannaworldnews/src/lib/articles.ts
export interface Article {
  slug: string; title: string; dek: string;
  region: string; date: string; author: string;
  html: string; // marked-rendered body
}
export const REGIONS: { slug: string; label: string }[]; // 4 fixed regions
export const articles: Article[];               // sorted newest-first
export function getArticle(slug: string): Article | undefined;
export function articlesByRegion(region: string): Article[]; // newest-first
```

entries() patterns:
```ts
// articles/[slug]/+page.ts
export const entries = () => articles.map((a) => ({ slug: a.slug }));
// region/[region]/+page.ts
export const entries = () => REGIONS.map((r) => ({ region: r.slug }));
```
